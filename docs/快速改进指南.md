### 4. 使用glfwWaitEventsTimeout（10分钟）

**文件**：`src/app/Application.cpp`

找到主循环，修改为：
```cpp
int Application::run()
{
    if (!impl_ || impl_->window == nullptr || !impl_->context) {
        return 1;
    }

    while (!impl_->window->shouldClose()) {
        if (impl_->needsRedraw) {
            impl_->window->pollEvents();
            impl_->renderFrame();
            impl_->needsRedraw = false;
        } else {
            // 等待事件，最多8ms超时，比sleep更省电
            glfwWaitEventsTimeout(0.008);
        }
    }

    return 0;
}
```

### 7. Measure结果缓存（45分钟）

**文件**：`include/tinalux/ui/Widget.h`

**步骤1**：添加缓存结构：
```cpp
class Widget {
protected:
    struct MeasureCache {
        Constraints constraints;
        core::Size result;
        bool valid = false;
        
        bool matches(const Constraints& c) const {
            if (!valid) return false;
            return constraints.minWidth() == c.minWidth() &&
                   constraints.maxWidth() == c.maxWidth() &&
                   constraints.minHeight() == c.minHeight() &&
                   constraints.maxHeight() == c.maxHeight();
        }
    };
    
    mutable MeasureCache measureCache_;
};
```

**步骤2**：修改`src/ui/Widget.cpp`中的measure：
```cpp
core::Size Widget::measure(const Constraints& constraints)
{
    if (measureCache_.matches(constraints)) {
        return measureCache_.result;
    }
    
    auto result = onMeasure(constraints);
    measureCache_ = {constraints, result, true};
    return result;
}
```

**步骤3**：在`markLayoutDirty()`中失效缓存：
```cpp
void Widget::markLayoutDirty()
{
    measureCache_.valid = false;
    if (layoutDirty_) {
        return;
    }
    layoutDirty_ = true;
    if (parent_) {
        parent_->onChildLayoutDirty();
    }
}
```

### 8. Font缓存池（45分钟）

**新建文件**：`src/ui/FontCache.h`

```cpp
#pragma once
#include <unordered_map>
#include "include/core/SkFont.h"
#include "include/core/SkTypeface.h"

namespace tinalux::ui {

class FontCache {
public:
    static FontCache& instance();
    
    const SkFont& getFont(float fontSize);
    void clear();
    
private:
    FontCache() = default;
    std::unordered_map<float, SkFont> cache_;
};

}  // namespace tinalux::ui
```

**新建文件**：`src/ui/FontCache.cpp`

```cpp
#include "FontCache.h"

namespace tinalux::ui {

FontCache& FontCache::instance()
{
    static FontCache instance;
    return instance;
}

const SkFont& FontCache::getFont(float fontSize)
{
    auto it = cache_.find(fontSize);
    if (it != cache_.end()) {
        return it->second;
    }
    
    SkFont font(SkTypeface::MakeDefault(), fontSize);
    font.setEdging(SkFont::Edging::kAntiAlias);
    font.setSubpixel(true);
    
    return cache_.emplace(fontSize, std::move(font)).first->second;
}

void FontCache::clear()
{
    cache_.clear();
}

}  // namespace tinalux::ui
```

**A**: 检查以下几点：
1. 是否所有新增的头文件都已include
2. 命名空间是否正确
3. CMakeLists.txt是否已更新
4. 清理构建目录重新编译：`rm -rf build && cmake -B build`

### Q2: 如何验证性能改进？

**A**: 使用Debug HUD查看帧率：
1. 启动应用
2. 按F3打开Debug HUD
