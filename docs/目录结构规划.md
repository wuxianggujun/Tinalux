## CMake 配置

### 根 CMakeLists.txt

```cmake
cmake_minimum_required(VERSION 3.31)
project(Tinalux LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include(FetchContent)
FetchContent_Declare(glfw
    GIT_REPOSITORY https://github.com/glfw/glfw.git
    GIT_TAG        3.4
)
set(GLFW_BUILD_DOCS OFF CACHE BOOL "" FORCE)
set(GLFW_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(GLFW_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)
FetchContent_MakeAvailable(glfw)

include(cmake/SkiaConfig.cmake)
tinalux_enable_skia()

add_subdirectory(src)

add_executable(Tinalux main.cpp)
target_link_libraries(Tinalux PRIVATE Tinalux::App)
```

### Core（零依赖）

```cmake
add_library(TinaluxCore STATIC core.cpp events/Event.cpp)
add_library(Tinalux::Core ALIAS TinaluxCore)
target_include_directories(TinaluxCore PUBLIC ${PROJECT_SOURCE_DIR}/include)
target_compile_features(TinaluxCore PUBLIC cxx_std_20)
```

### Platform（Core + GLFW/NDK）

```cmake
add_library(TinaluxPlatform STATIC Window.cpp)
add_library(Tinalux::Platform ALIAS TinaluxPlatform)
target_include_directories(TinaluxPlatform PUBLIC ${PROJECT_SOURCE_DIR}/include)
target_link_libraries(TinaluxPlatform PUBLIC Tinalux::Core)

if(NOT ANDROID)
    target_sources(TinaluxPlatform PRIVATE glfw/GLFWWindow.cpp)
    target_link_libraries(TinaluxPlatform PRIVATE glfw)
    target_compile_definitions(TinaluxPlatform PRIVATE TINALUX_PLATFORM_GLFW)
endif()
```

### Rendering（Core + Skia，不链接 glfw 和 Platform）

```cmake
add_library(TinaluxRendering STATIC rendering.cpp GLContext.cpp WindowSurface.cpp)
add_library(Tinalux::Rendering ALIAS TinaluxRendering)
target_include_directories(TinaluxRendering PUBLIC ${PROJECT_SOURCE_DIR}/include)
target_link_libraries(TinaluxRendering
    PUBLIC Tinalux::Core Skia::Skia
)
```

### UI（Core + Skia，不链接 Rendering 和 Platform）

```cmake
add_library(TinaluxUI STATIC
    Constraints.cpp Widget.cpp Container.cpp
    layouts/VBoxLayout.cpp layouts/HBoxLayout.cpp
    widgets/Label.cpp widgets/Button.cpp widgets/Panel.cpp widgets/TextInput.cpp
    themes/Theme.cpp
)
add_library(Tinalux::UI ALIAS TinaluxUI)
target_include_directories(TinaluxUI PUBLIC ${PROJECT_SOURCE_DIR}/include)
target_link_libraries(TinaluxUI
    PUBLIC Tinalux::Core Skia::Skia
)
```

### App（唯一胶水层）

```cmake
add_library(TinaluxApp STATIC Application.cpp)
add_library(Tinalux::App ALIAS TinaluxApp)
target_include_directories(TinaluxApp PUBLIC ${PROJECT_SOURCE_DIR}/include)
target_link_libraries(TinaluxApp
    PUBLIC Tinalux::Platform Tinalux::Rendering Tinalux::UI Tinalux::Core
)
```

## 命名约定

| 类别 | 规则 | 示例 |
|------|------|------|
| 文件 | PascalCase | `Widget.h`, `GLFWWindow.cpp` |
| 类 | PascalCase | `Container`, `VBoxLayout` |
| 函数 | camelCase | `addChild()`, `hitTest()` |
| 成员变量 | camelCase_ | `bounds_`, `visible_` |
| 命名空间 | lowercase | `tinalux::platform` |
